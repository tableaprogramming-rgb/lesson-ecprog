#include <stdio.h>

int main() {
    int year = 2000;

    if (year % 4 == 0 || year % 100 != 0 || year % 400 == 0) {
        printf("Leap year\n");
    } else {
        printf("Not a leap year\n");

  5. Verify: Test your fixed condition with all 4 cases
